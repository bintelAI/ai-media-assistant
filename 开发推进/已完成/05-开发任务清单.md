export async function sendMessage<T, R>(
  type: MessageType,
  data?: T

  const response = await chrome.runtime.sendMessage({
    type,
    data
  })

// src/entrypoints/background/index.ts

chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
  handleMessage(message, sender)
    .then(result => sendResponse(result))
    .catch(err => sendResponse({ success: false, error: err.message }))
  
  return true  // 保持消息通道开启
})

async function handleMessage(message: Message, sender: chrome.runtime.MessageSender) {
  switch (message.type) {
    case 'collect:post':
      return collectHandler.handlePost(message.data)
    
    case 'export:data':
      return exportHandler.handleExport(message.data)
    
    default:
      return { success: false, error: 'Unknown message type' }
  }
}

export function detectXHSPage(url: string): PageType {
  // 帖子详情页
  if (url.match(/\/explore\/[\w]+/)) {
    return 'post_detail'
  }
  
  // 作者主页
  if (url.match(/\/user\/profile\/[\w]+/)) {
    return 'author_profile'
  }
  
  // 搜索结果
  if (url.includes('/search_result/')) {
    return 'search_result'
  }
  
  // 首页
  if (url === 'https://www.xiaohongshu.com' || url.includes('/explore?type=')) {
    return 'feed_list'
  }
  
  return 'unknown'
}

// 监听 URL 变化
const observer = new MutationObserver(() => {

export async function extractPost(): Promise<ExtractResult<PostEntity>> {
  // 策略 1：从 DOM 抽取
  const domData = extractFromDOM()
  
  if (domData.success) {
    return domData
  }
  
  // 策略 2：从页面状态抽取
  const stateData = extractFromPageState()
  
  if (stateData.success) {
    return stateData
  }
  
  // 策略 3：从网络请求拦截抽取（可选）

#### 2. 从页面状态抽取

小红书通常会在 window 上暴露状态：

```typescript
function extractFromPageState(): ExtractResult<PostEntity> {
  try {
    // 尝试从 __INITIAL_STATE__ 获取
    const state = (window as any).__INITIAL_STATE__
    
    if (state?.note?.noteDetailMap) {
      const noteData = Object.values(state.note.noteDetailMap)[0]
      
      return {
        success: true,
        data: {
          platform: 'xhs',
          postId: noteData.noteId,
          title: noteData.title,
          content: noteData.desc,
          authorId: noteData.user.userId,
          authorName: noteData.user.nickname,
          likeCount: noteData.interactInfo?.likeCount,
          commentCount: noteData.interactInfo?.commentCount,
          collectCount: noteData.interactInfo?.collectCount
        }
      }
    }
    
    return { success: false, error: '页面状态不存在' }
  } catch (err) {
    return { success: false, error: err.message }
  }
}
```

```typescript
function parseChineseNumber(text: string): number | undefined {
  if (!text) return undefined
  
  text = text.trim()
  
  if (text.includes('万')) {
    const num = parseFloat(text.replace('万', ''))
    return num * 10000
  }
  
  if (text.includes('亿')) {
    const num = parseFloat(text.replace('亿', ''))
    return num * 100000000
  }
  
  return parseInt(text.replace(/[^\d]/g, ''), 10)
}
```

function createButton(text: string, onClick: () => void): HTMLElement {
  const container = document.createElement('div')
  const shadow = container.attachShadow({ mode: 'open' })
  
  shadow.innerHTML = `
    <style>
      button {
        padding: 8px 16px;
        background: #ff2442;
        color: white;
        border: none;
        border-radius: 4px;
        cursor: pointer;
      }
      button:disabled {
        opacity: 0.6;
        cursor: not-allowed;
      }
    </style>
    <button>${text}</button>
  `
  
  const button = shadow.querySelector('button')
  button.addEventListener('click', onClick)
  
  return container

# 开发风险提示

---

## 1. 平台稳定性风险

小红书、抖音、快手页面结构可能变化，导致抽取失败。

**应对**

- 多策略抽取（DOM + 页面状态 + 网络拦截）
- 定期维护适配器
- 建立失败监控

---

## 2. Chrome API 权限风险

Chrome Web Store 审核可能对权限敏感。

**应对**

- 权限最小化
- 只申请必要 host_permissions
- 提供清晰的权限说明文案

---

## 3. IndexedDB 性能风险

数据量大时可能卡顿。

**应对**

- 分页查询
- 虚拟滚动
- 定期清理旧数据

---

## 4. 导出内存风险

导出大量数据时可能内存溢出。

**应对**

- 分批导出
- 流式生成文件（如果可能）
- 限制单次导出数量
